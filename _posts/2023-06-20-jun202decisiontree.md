```python
# 이 데이터를 훈련용, 테스트용으로 나눠서, 시험지 / 정답지로 만들기 
x = cancer.data
y = cancer.target

x_train, x_test, y_train, y_test = train_test_split(x,y,test_size=0.2,
                                                    random_state=1024, # seed의 개념
                                                    stratify=cancer.target
                                                    )
# 구성비가 다른 불균형 상황에서는 stratify 옵션 넣어주기 
# train, test 데이터에 있는 label 값의 비율이 서로 다르면
# 학습결과나 모델 평가 등에서 잘못된 영향을 끼칠 가능성이 있음 
# dataset쪽의 클래스의 비율과 같은 비율로 조정 
```

```python
# 의사결정나무 모델 생성 (깊이 = 4)
cancer_tree = DecisionTreeClassifier(max_depth=4)
```

```python
# 모델 학습
cancer_tree.fit(x,y)
```

# 정확도 확인
